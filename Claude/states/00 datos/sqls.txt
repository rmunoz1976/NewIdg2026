NECESITO AHORA EXCLUIR LOS TIPOS AMD1, ADM2, ADM3, ADM4 Y ADM5
Y SEPARALOS RESTANTES EN UN ARCHIVO SEPARADO

$inputFolder = "C:\dev\idg\docs\Claude\states\ORIGINAL"
$outputFolder = "C:\dev\idg\docs\Claude\states\ADM5"

New-Item -ItemType Directory -Force -Path $outputFolder | Out-Null

Get-ChildItem -Path $inputFolder -Filter "*.txt" | ForEach-Object {
    $countryCode = $_.BaseName
    $outputFile = Join-Path $outputFolder ($countryCode + "_ADM5.txt")
    
    $reader = [System.IO.StreamReader]::new($_.FullName)
    $writer = [System.IO.StreamWriter]::new($outputFile)
    $count = 0
    
    while (($line = $reader.ReadLine()) -ne $null) {
        $cols = $line -split "`t"
        if ($cols.Count -ge 9 -and $cols[6] -eq "A" -and $cols[7] -eq "ADM5" -and $cols[8] -eq $countryCode) {
            $writer.WriteLine($line)
            $count++
        }
    }
    $reader.Close()
    $writer.Close()
    
    Write-Host "$countryCode : $count registros ADM5"
}

Write-Host "Proceso completado"

SELECT c.name_english AS pais, prefijo,  s.cantidad
FROM (
    SELECT  SUBSTRING_INDEX(code, '-', 1) AS prefijo,  COUNT(*) AS cantidad
    FROM intermediary_db.t076t_states
    GROUP BY SUBSTRING_INDEX(code, '-', 1)
) s
LEFT JOIN intermediary_db.t074t_countries c ON c.code = s.prefijo
ORDER BY prefijo desc;
