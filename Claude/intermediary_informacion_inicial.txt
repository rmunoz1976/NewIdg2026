== Cifrado simétrico AES (Advanced Encryption Standard) en modo CBC ==

- Datos de envío de Java deben enviarse encriptados y se deben desencriptar en Dart

Dart
import 'package:encrypt/encrypt.dart';

class AESCrypto {
  static final String keyString = 'my32lengthsupersecretnooneknows1';
  static final String ivString = 'my16lengthivkey1';

  // Cifra un campo individual y devuelve su versión en base64
  static String encryptField(dynamic field) {
    final key = Key.fromUtf8(keyString);
    final iv = IV.fromUtf8(ivString);
    final encrypter = Encrypter(AES(key, mode: AESMode.cbc));
    
    // Convertimos el campo a String (funciona para números, booleanos, etc.)
    final fieldString = field.toString();
    return encrypter.encrypt(fieldString, iv: iv).base64;
  }

  // Descifra un campo individual desde base64
  static dynamic decryptField(String encryptedField) {
    final key = Key.fromUtf8(keyString);
    final iv = IV.fromUtf8(ivString);
    final encrypter = Encrypter(AES(key, mode: AESMode.cbc));
    
    return encrypter.decrypt64(encryptedField, iv: iv);
  }

  // Cifra todos los campos de un mapa individualmente
  static Map<String, String> encryptMap(Map<String, dynamic> data) {
    final encryptedMap = <String, String>{};
    
    data.forEach((key, value) {
      encryptedMap[key] = encryptField(value);
    });
    
    return encryptedMap;
  }

  // Descifra todos los campos de un mapa individualmente
  static Map<String, dynamic> decryptMap(Map<String, String> encryptedData) {
    final decryptedMap = <String, dynamic>{};
    
    encryptedData.forEach((key, value) {
      decryptedMap[key] = decryptField(value);
    });
    
    return decryptedMap;
  }
}
