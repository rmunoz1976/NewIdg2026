# ============================================================
# Todos los códigos de nivel a procesar
# ============================================================
$allCodes = @(
    "ADM1","ADM2","ADM3","ADM4","ADM5",
    "ADM1H","ADM2H","ADM3H","ADM4H","ADM5H",
    "ADMD","ADMDH","PRSH","ZN","TERR",
    "PCLI","PCLD","PCLIX","PCLF","PCLH","PCLS"
)

# ============================================================
# Crear todas las carpetas de antemano
# ============================================================
foreach ($code in $allCodes) {
    New-Item -ItemType Directory -Force -Path (Join-Path $baseOutputFolder $code) | Out-Null
}
Write-Host "Carpetas creadas: $($allCodes.Count)"
Write-Host ""

# ============================================================
# Procesar cada archivo .txt de la carpeta ORIGINAL
# ============================================================
$totalFiles   = 0
$totalRecords = 0
$skipped      = 0

Get-ChildItem -Path $inputFolder -Filter "*.txt" | Sort-Object Name | ForEach-Object {
    $countryCode = $_.BaseName

    # Obtener nombre del país
    if ($countryNames.ContainsKey($countryCode)) {
        $countryName = $countryNames[$countryCode]
    } else {
        $countryName = $countryCode.ToLower()
        Write-Host "  AVISO: $countryCode no está en el mapa de países — usando código como nombre" -ForegroundColor Yellow
    }

    $reader = [System.IO.StreamReader]::new($_.FullName, [System.Text.Encoding]::UTF8)

    # Agrupar líneas por feature_code
    $grupos = @{}
    while (($line = $reader.ReadLine()) -ne $null) {
        $cols = $line -split "`t"
        if ($cols.Count -ge 9 -and $cols[6] -eq "A" -and $cols[8] -eq $countryCode) {
            $featureCode = $cols[7]
            if ($allCodes -contains $featureCode) {
                if (-not $grupos.ContainsKey($featureCode)) {
                    $grupos[$featureCode] = [System.Collections.Generic.List[string]]::new()
                }
                $grupos[$featureCode].Add($line)
            }
        }
    }
    $reader.Close()

    if ($grupos.Count -eq 0) {
        Write-Host "$countryCode : sin registros" -ForegroundColor DarkGray
        $skipped++
    } else {
        # Escribir cada grupo en su carpeta
        foreach ($featureCode in ($grupos.Keys | Sort-Object)) {
            $outputFolder = Join-Path $baseOutputFolder $featureCode
            $fileName     = $countryCode + "_" + $countryName + "_" + $featureCode + ".txt"
            $outputFile   = Join-Path $outputFolder $fileName
            $writer = [System.IO.StreamWriter]::new($outputFile, $false, [System.Text.Encoding]::UTF8)
            foreach ($line in $grupos[$featureCode]) {
                $writer.WriteLine($line)
            }
            $writer.Close()
            $count = $grupos[$featureCode].Count
            $totalRecords += $count
            Write-Host "  $fileName : $count registros"
        }
    }
    $totalFiles++
}

# ============================================================
# Resumen final
# ============================================================
Write-Host ""
Write-Host "================================================" -ForegroundColor Cyan
Write-Host "Proceso completado" -ForegroundColor Cyan
Write-Host "Archivos procesados : $totalFiles"
Write-Host "Sin registros       : $skipped"
Write-Host "Registros totales   : $totalRecords"
Write-Host "================================================" -ForegroundColor Cyan
